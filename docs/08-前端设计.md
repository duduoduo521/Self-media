## 3. XSS 防护

AI 生成的内容直接展示给用户，需防止 XSS：

### 3.1 策略

- 所有 AI 生成内容在渲染前通过 DOMPurify sanitize
- 使用 `v-text` 而非 `v-html` 渲染纯文本内容
- 必须使用 `v-html` 时，配合 DOMPurify 过滤

### 3.2 实现

```typescript
// utils/sanitize.ts
import DOMPurify from 'dompurify';

export function sanitizeHtml(dirty: string): string {
  return DOMPurify.sanitize(dirty, {
    ALLOWED_TAGS: ['b', 'i', 'em', 'strong', 'p', 'br', 'ul', 'ol', 'li'],
    ALLOWED_ATTR: [],
  });
}

export function sanitizeText(text: string): string {
  // 纯文本场景：去除所有 HTML 标签
  return DOMPurify.sanitize(text, {
    ALLOWED_TAGS: [],
    ALLOWED_ATTR: [],
  });
}
```

```vue
<!-- 组件中使用 -->
<template>
  <!-- 纯文本渲染：使用 v-text -->
  <div v-text="aiGeneratedContent" />

  <!-- 必须渲染 HTML 时：sanitize -->
  <div v-html="sanitizeHtml(aiGeneratedContent)" />
</template>
```

## 5. 双端适配层

前端需同时支持 Tauri（桌面）和 HTTP（Web）两种通信方式：

```typescript
// utils/api.ts

interface ApiClient {
  post<T>(path: string, data: any): Promise<T>;
  get<T>(path: string): Promise<T>;
  put<T>(path: string, data: any): Promise<T>;
  delete<T>(path: string): Promise<T>;
}

// Tauri 桌面端实现
class TauriApiClient implements ApiClient {
  async post<T>(path: string, data: any): Promise<T> {
    const command = pathToCommand(path, 'post');
    return await invoke(command, data);
  }
  // ...
}

// Web 端 HTTP 实现
class HttpApiClient implements ApiClient {
  private baseUrl: string;

  constructor() {
    this.baseUrl = import.meta.env.VITE_API_URL || 'http://localhost:3000';
  }

  async post<T>(path: string, data: any): Promise<T> {
    const response = await axios.post(`${this.baseUrl}${path}`, data, {
      withCredentials: true,  // 携带 Cookie
      headers: {
        'X-CSRF-Token': getCsrfToken(),  // CSRF 防护
      },
    });
    return response.data;
  }
  // ...
}

// 自动选择实现
export const api: ApiClient = window.__TAURI_INTERNALS__
  ? new TauriApiClient()
  : new HttpApiClient();
```

## 6. SSE 流式文本展示

AI 文本生成支持流式输出，前端需实时渲染：

```typescript
// composables/useSSE.ts
import { ref } from 'vue';

export function useSSE() {
  const content = ref('');
  const isStreaming = ref(false);

  async function startStream(taskId: string) {
    isStreaming.value = true;
    content.value = '';

    // Tauri 端通过事件监听
    if (window.__TAURI_INTERNALS__) {
      const { listen } = await import('@tauri-apps/api/event');
      const unlisten = await listen<string>('text-stream', (event) => {
        content.value += event.payload;
      });
      return unlisten;
    }

    // Web 端通过 EventSource
    const eventSource = new EventSource(`/api/tasks/${taskId}/stream`);
    eventSource.onmessage = (event) => {
      const chunk = JSON.parse(event.data);
      content.value += chunk.delta?.content || '';
    };
    eventSource.onerror = () => {
      isStreaming.value = false;
      eventSource.close();
    };
  }

  return { content, isStreaming, startStream };
}
```
