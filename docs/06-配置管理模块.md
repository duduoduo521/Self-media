## 1. 功能清单

| 功能 | 说明 | 优先级 |
|------|------|--------|
| API Key 管理 | 设置/获取 MiniMax API Key（加密存储） | P0 |
| 平台配置管理 | 启用/禁用平台、配图数量、Cookie 存储 | P0 |
| 平台登录态管理 | Cookie 更新、健康检查 | P0 |

## 2. 数据模型

### 2.1 表结构

```sql
CREATE TABLE platform_configs (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id     INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    platform    TEXT NOT NULL,
    enabled     INTEGER NOT NULL DEFAULT 1,
    image_count INTEGER NOT NULL DEFAULT 3,
    cookies     TEXT,               -- AES-256-GCM 加密存储
    extra       TEXT,               -- JSON，平台特定配置
    created_at  TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at  TEXT NOT NULL DEFAULT (datetime('now')),
    UNIQUE(user_id, platform)
);

CREATE TABLE api_keys (
    id            INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id       INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    provider      TEXT NOT NULL,         -- 'minimax'
    encrypted_key TEXT NOT NULL,         -- AES-256-GCM 加密存储
    region        TEXT NOT NULL DEFAULT 'cn',  -- 'cn' | 'global'
    created_at    TEXT NOT NULL DEFAULT (datetime('now')),
    UNIQUE(user_id, provider)
);

CREATE TABLE user_preferences (
    user_id         INTEGER PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    default_mode    TEXT NOT NULL DEFAULT 'text',  -- 'text' | 'video'
    default_tags    TEXT NOT NULL DEFAULT '[]',     -- JSON array
    auto_publish    INTEGER NOT NULL DEFAULT 0,     -- 生成后自动发布
    updated_at      TEXT NOT NULL DEFAULT (datetime('now'))
);
```

## 3. 配置服务

```rust
pub struct ConfigService {
    db: SqlitePool,
}

impl ConfigService {
    pub fn new(db: SqlitePool) -> Self {
        Self { db }
    }

    // ---- API Key 管理 ----

    /// 设置 API Key（加密存储）
    pub async fn set_api_key(
        &self,
        user_id: i64,
        provider: &str,
        key: &str,
        region: &str,
        user_key: &UserKey,
    ) -> Result<(), AppError> {
        // 1. 校验 API Key 格式（非空，合理长度）
        if key.trim().is_empty() {
            return Err(AppError::validation(INPUT_001, "API Key 不能为空"));
        }

        // 2. 加密存储
        let encrypted_key = user_key.encrypt(key)?;

        // 3. Upsert
        sqlx::query!(
            "INSERT INTO api_keys (user_id, provider, encrypted_key, region) \
             VALUES (?, ?, ?, ?) \
             ON CONFLICT(user_id, provider) DO UPDATE SET \
             encrypted_key = excluded.encrypted_key, \
             region = excluded.region",
            user_id, provider, encrypted_key, region
        )
        .execute(&self.db)
        .await?;

        Ok(())
    }

    /// 获取解密后的 API Key
    pub async fn get_api_key(
        &self,
        user_id: i64,
        provider: &str,
        user_key: &UserKey,
    ) -> Result<(String, MiniMaxRegion), AppError> {
        let row = sqlx::query!(
            "SELECT encrypted_key, region FROM api_keys WHERE user_id = ? AND provider = ?",
            user_id, provider
        )
        .fetch_optional(&self.db)
        .await?
        .ok_or(AppError::config(CONFIG_001, &format!("未配置 {} API Key", provider)))?;

        let decrypted_key = user_key.decrypt(&row.encrypted_key)?;
        let region = match row.region.as_str() {
            "global" => MiniMaxRegion::Global,
            _ => MiniMaxRegion::CN,
        };
        Ok((decrypted_key, region))
    }

    // ---- 平台配置管理 ----

    /// 设置平台配置
    pub async fn set_platform_config(
        &self,
        user_id: i64,
        platform: &Platform,
        enabled: bool,
        image_count: u32,
        cookies: Option<&str>,
        user_key: &UserKey,
    ) -> Result<(), AppError> {
        let platform_str = serde_json::to_string(platform)?;
        let encrypted_cookies = match cookies {
            Some(c) if !c.is_empty() => Some(user_key.encrypt(c)?),
            _ => None,
        };

        sqlx::query!(
            "INSERT INTO platform_configs (user_id, platform, enabled, image_count, cookies) \
             VALUES (?, ?, ?, ?, ?) \
             ON CONFLICT(user_id, platform) DO UPDATE SET \
             enabled = excluded.enabled, \
             image_count = excluded.image_count, \
             cookies = COALESCE(excluded.cookies, platform_configs.cookies), \
             updated_at = datetime('now')",
            user_id, platform_str, enabled as i32, image_count as i32, encrypted_cookies
        )
        .execute(&self.db)
        .await?;

        Ok(())
    }

    /// 获取用户所有平台配置
    pub async fn get_platform_configs(
        &self,
        user_id: i64,
    ) -> Result<Vec<PlatformConfig>, AppError> {
        let rows = sqlx::query!(
            "SELECT platform, enabled, image_count, cookies, extra FROM platform_configs WHERE user_id = ?",
            user_id
        )
        .fetch_all(&self.db)
        .await?;

        let mut configs = Vec::new();
        for row in rows {
            let platform: Platform = serde_json::from_str(&row.platform)?;
            configs.push(PlatformConfig {
                platform,
                enabled: row.enabled != 0,
                image_count: row.image_count as u32,
                cookies: row.cookies,  // 保持加密状态，使用时再解密
                extra: row.extra
                    .map(|s| serde_json::from_str(&s).unwrap_or_default())
                    .unwrap_or_default(),
            });
        }
        Ok(configs)
    }

    /// 获取用户已启用的平台列表
    pub async fn get_enabled_platforms(
        &self,
        user_id: i64,
    ) -> Result<Vec<Platform>, AppError> {
        let rows = sqlx::query!(
            "SELECT platform FROM platform_configs WHERE user_id = ? AND enabled = 1",
            user_id
        )
        .fetch_all(&self.db)
        .await?;

        rows.iter()
            .map(|r| serde_json::from_str(&r.platform).map_err(AppError::from))
            .collect()
    }

    /// 解密平台 Cookie
    pub fn decrypt_cookies(
        &self,
        encrypted: &str,
        user_key: &UserKey,
    ) -> Result<String, AppError> {
        user_key.decrypt(encrypted)
    }

    // ---- 用户偏好 ----

    pub async fn get_preferences(&self, user_id: i64) -> Result<UserPreferences, AppError> {
        let row = sqlx::query!(
            "SELECT default_mode, default_tags, auto_publish FROM user_preferences WHERE user_id = ?",
            user_id
        )
        .fetch_optional(&self.db)
        .await?;

        match row {
            Some(r) => Ok(UserPreferences {
                default_mode: match r.default_mode.as_str() {
                    "video" => TaskMode::Video,
                    _ => TaskMode::Text,
                },
                default_tags: serde_json::from_str(&r.default_tags).unwrap_or_default(),
                auto_publish: r.auto_publish != 0,
            }),
            None => Ok(UserPreferences {
                default_mode: TaskMode::Text,
                default_tags: vec![],
                auto_publish: false,
            }),
        }
    }

    pub async fn set_preferences(&self, user_id: i64, prefs: &UserPreferences) -> Result<(), AppError> {
        let mode_str = match prefs.default_mode {
            TaskMode::Text => "text",
            TaskMode::Video => "video",
        };
        sqlx::query!(
            "INSERT INTO user_preferences (user_id, default_mode, default_tags, auto_publish) \
             VALUES (?, ?, ?, ?) \
             ON CONFLICT(user_id) DO UPDATE SET \
             default_mode = excluded.default_mode, \
             default_tags = excluded.default_tags, \
             auto_publish = excluded.auto_publish, \
             updated_at = datetime('now')",
            user_id, mode_str, serde_json::to_string(&prefs.default_tags)?, prefs.auto_publish as i32
        )
        .execute(&self.db)
        .await?;
        Ok(())
    }
}
```

## 4. 用户密钥缓存

Web 端多用户场景下，需要缓存各用户的 `UserKey`，避免每次请求都重新派生：

```rust
use std::sync::Arc;
use tokio::sync::RwLock;
use lru::LruCache;

/// 用户密钥缓存（Web 端使用）
pub struct UserKeyCache {
    cache: RwLock<LruCache<i64, UserKey>>,
}

impl UserKeyCache {
    pub fn new(max_capacity: usize) -> Self {
        Self {
            cache: RwLock::new(LruCache::new(max_capacity.try_into().unwrap())),
        }
    }

    /// 获取用户密钥，若缓存未命中则从数据库加载并派生
    pub async fn get_or_derive(
        &self,
        user_id: i64,
        db: &SqlitePool,
        password: Option<&str>,
    ) -> Result<Arc<UserKey>, AppError> {
        // 1. 检查缓存
        {
            let cache = self.cache.read().await;
            if let Some(key) = cache.get(&user_id) {
                return Ok(Arc::new(key.clone()));
            }
        }

        // 2. 需要密码来派生密钥
        let password = password.ok_or(AppError::crypto(CRYPTO_001, "用户密钥未缓存，需重新登录"))?;

        // 3. 从数据库获取 salt
        let salt = sqlx::query_scalar!(
            "SELECT salt FROM users WHERE id = ?",
            user_id
        )
        .fetch_one(db)
        .await?;

        // 4. 派生密钥
        let user_key = UserKey::derive_from_password(password, &salt)?;

        // 5. 写入缓存
        {
            let mut cache = self.cache.write().await;
            cache.put(user_id, user_key.clone());
        }

        Ok(Arc::new(user_key))
    }

    /// 用户修改密码或登出时清除缓存
    pub async fn invalidate(&self, user_id: i64) {
        let mut cache = self.cache.write().await;
        cache.pop(&user_id);
    }
}
```
